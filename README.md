<h1>FLAT GALLERY - simple jQuery Gallery with Flat Colors</h1>
<h3>How to use:</h3>
<pre>
	var myImages = [
						...

						{
                            file: 'big image URL',
                            thumb: 'thumb image URL',
                            size: 'image size ',
                            mime: 'image mime type'    
                        }
                        
                        ...

                    ];
	$(element).flatGallery(myImages);
</pre>
